## ReplicaSet

apiVersion: apps/v1
kind: ReplicaSet
metadata:
    name: task-nginx
    labels:
        env: task-app
spec:
    replicas: 3
    selector:
        matchLabels:
            env: task-app
    template:
        metadata:
            labels:
                env: task-app
        spec:
            containers:
                - name: nginx
                  image: nginx
    
    
---


## Deployment

apiVersion: apps/v1
kind: Deployment
metadata: 
  name: nginx
  labels:
    tier: backend
spec:
  replicas: 3
  selector:
    matchLabels:
      app: v1       
  template:
    metadata:
      labels:
        app: v1     
    spec:
      containers:
        - name: nginx
          image: nginx:1.23.0

            
kubectl set image deploy/nginx nginx=nginx:1.23.4


kubectl annotate deploy/nginx   kubernetes.io/change-cause="Pick up patch version"

kubectl rollout undo deployment/nginx --to-revision=1
